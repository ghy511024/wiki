后台弹窗

1.引入插件

            <link rel="stylesheet" href="css/zyalert.css"><script type="text/javascript" src="../lib/dist/zlib.min.js"></script>

2.使用插件

            
<!--添加一个div，类名自定义，例如haha；并设置style="display: none"-->
<div class="zyinfo" style="display: none">
    <!--标题-->
    <div class="zy-title">修改运营信息</div>
    <!--文本框-->
    <div class="item">
        <label for="">章鱼uid</label>
        <input type="text">
    </div>
    <!--按钮-->
    <div class="zy-btn">
        <div class="btn btn-center submit">确定</div>
    </div>
</div>
<script>
    <!--传入自定义的类名，比如：haha-->
    var panel = new zlib.admin.zpanel($('.haha'))
<script/>
             
显示弹框

3.调用方法

Show()----显示弹窗

panel.Show();

Hide()----隐藏弹窗

panel.Hide();

submit(传入一个提交信息的函数)----提交弹窗信息,默认在点击类名为submit的按钮后执行传入的函数;

panel.submit(function () {
        console.log("弹窗信息已提交");
    });

cancel()----默认在点击类名为cancel的按钮后执行;

panel.cancel();
