章鱼滚动条插件

1.引入插件


      <script type="text/javascript" src="dist/zlib.js"></script>
      <script type="text/javascript" src="dist/zlib-plug-scroll.js"></script>

2.使用插件


 <div class="aa" style="width:300px;height:200px;overflow: hidden;">
     <div class="sc_content"  style="width:300px;height:600px;background: -webkit-linear-gradient(top,#CE20B5,#56DBE2);">  </div>
 </div>
<script type="text/javascript">
    $(function () {
        var sc = new zlib.plug.zscroll($(".aa"))// 需要new （页面可能同时存在多个插件）
    }
</script>

3.效果

4.参数支持


<script type="text/javascript">
    $(function () {
        var sc = new zlib.plug.zscroll($(".aa"), {
            scWidth: "20px",        //整体滚动条的宽度，可不配置，默认10px
            barWidth: "10px",       //滚动条宽度，可不配置，默认4px
            barBgColor:"blue",      //滚动条颜色，可不配置，默认#a6b0c3
            barHoverColor: "#ff3346",       //hover时滚动条颜色，可不配置，默认##caccd0
            panelBgColor:"rgba(0, 0, 0, 0.5)"       //可滚动区域背景色，可不配置，默认透明
        })
        //如仅需要展示滚动条，但对于颜色宽度都没有过多要求，可直接引入如下代码
        var sc = new zlib.plug.zscroll($(".aa"));
    })
</script>

5.公有方法 update


<script type="text/javascript">
    $(function () {
        var sc = new zlib.plug.zscroll($(".aa")); 
        var app = $("<div style='background: yellow;height:200px'></div>") 
        $("button").click(function() {
            $(".sc_content").append(app);
            sc.update();
        })
        //当有需求向内容区append内容时，需要添加事件时，触发update函数
        //demo为 添加update和没有添加update的两种情况
    })
</script>

6.公有方法 scrollTo


<script type="text/javascript">
    $(function () {
        var sc = new zlib.plug.zscroll($(".aa")); 
        $("button").click(function() {
            sc3.scrollTo("bottom")
        })
        //点击后设置滚动条的位置，支持关键字top，bottom，和任意数值
        //当关键字设置为"top"时，滚动条初始位置是在顶部；设置为"bottom"时，位置在底部；设置为数值50时，距离顶部50px
    })
</script>
