章鱼js ajax组件化

通用ajax 组件，去除重复代码，避免业务处理逻辑写在 ajax返回中。 


        var model = {
            send: function () {
                model.ajax("http://www.abc/xx", {"param": "2"}).then(function (ret) {
                    console.log("请求成功")
                }, function () {
                    console.log("请求失败")
                })
            },
            ajax: function (url, data) {
                var dfd = $.Deferred();
                $.ajax({
                    url: url,
                    type: "post",
                    data: (data),
                    dataType: "json",
                    error: function (jqXHR, textStatus, errorThrown) {
                        dfd.reject();
                    },
                    success: function (ret) {
                        dfd.resolve(ret);
                    }
                });
                return dfd.promise();
            }
        }
